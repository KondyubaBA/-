### Фабричный метод

#### Продукты
```cs
// Abstract product interface
public interface IProduct
{
    string GetName();
}

// Concrete product classes
public class ConcreteProductA : IProduct
{
    public string GetName() => "Product A"; 
}

public class ConcreteProductB : IProduct
{
    public string GetName() => "Product B";    
}
```

#### Фабрики
```cs
// Abstract factory interface
public interface IFactory
{
    IProduct CreateProduct();
}

// Concrete factory classes
public class ConcreteFactoryA : IFactory
{
    public IProduct CreateProduct() => new ConcreteProductA(); 
}

public class ConcreteFactoryB : IFactory
{
    public IProduct CreateProduct() => new ConcreteProductB();    
}
```

#### Клиент код
```cs
// Client code
public class Client
{
    public void DoSomething(IFactory factory)
    {
        IProduct product = factory.CreateProduct();
        Console.WriteLine(product.GetName());
    }
}
```

#### Использование
```cs
// Usage example
public class Program
{
    public static void Main()
    {
        Client client = new Client();

        // Create an instance of ConcreteFactoryA
        IFactory factoryA = new ConcreteFactoryA();
        client.DoSomething(factoryA); // Output: Product A

        // Create an instance of ConcreteFactoryB
        IFactory factoryB = new ConcreteFactoryB();
        client.DoSomething(factoryB); // Output: Product B
    }
}
```

#### Пример
```cs
public abstract class Pizza
{
    public abstract string Prepare();
}

public class PepperoniPizza : Pizza
{
    public override string Prepare()
    {
        return "Preparing pepperoni pizza...";
    }
}

public class MargheritaPizza : Pizza
{
    public override string Prepare()
    {
        return "Preparing Margherita pizza...";
    }
}

public abstract class PizzaStore
{
    public abstract Pizza CreatePizza();

    public void OrderPizza()
    {
        Pizza pizza = CreatePizza();
        Console.WriteLine(pizza.Prepare());
    }
}

public class NewYorkPizzaStore : PizzaStore
{
    public override Pizza CreatePizza()
    {
        return new PepperoniPizza();
    }
}

public class ChicagoPizzaStore : PizzaStore
{
    public override Pizza CreatePizza()
    {
        return new MargheritaPizza();
    }
}

class Program
{
    static void Main(string[] args)
    {
        PizzaStore nyStore = new NewYorkPizzaStore();
        nyStore.OrderPizza();

        PizzaStore chiStore = new ChicagoPizzaStore();
        chiStore.OrderPizza();
    }
}
```
